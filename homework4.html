$l(\beta)=\sum_{i=1}^N\left\{y_ilogp(x_i;\beta)+(1-y_i)log(1-p(x_i;\beta))\right\}=y_ix_i^T\beta-y_ilog(1+e^{x_i^T\beta})$

So, $\frac{\partial ^2l(\beta)}{\partial \beta \partial \beta^T}=\sum_{i=1}^N-x_i\frac{\partial p_i}{\partial \beta^T}=\sum_{i=1}^N\frac{-x_ix_i^Te^{x_i^T\beta}}{(1+e^{x_i^T\beta})^2}=\sum_{i=1}^N-p_i(1-p_i)x_ix_i^T(p_i=p(x_i;\beta))$

import matplotlib.pyplot as plt
import numpy as np
import pandas as pd

def generate_data(N, true_beta):
    X_1 = np.random.normal(size=N)
    X_2 = np.random.normal(size=N)
    X = np.column_stack((np.ones(N), X_1, X_2))
    p = 1 / (1 + np.exp(-X @ true_beta))
    y = np.random.binomial(1, p)

    return X, y 

def newton(X, y, times=30, tolerence=1e-5):
    beta = np.zeros(X.shape[1])
    for _ in range(times):
        p_hat = 1 / (1 + np.exp(-X @ beta))
        partial_one = X.T @ (y - p_hat)
        partial_two = -X.T @ np.diag(p_hat * (1 - p_hat)) @ X
        beta_new = beta - np.linalg.solve(partial_two, partial_one)

        if np.max(np.abs(beta_new - beta)) < tolerence:
            beta = beta_new
            break
        beta = beta_new
    
    return beta

from sklearn.model_selection import KFold

def ridge(X, y, lambda_para):
    n = X.shape[1]
    beta_ridge = np.linalg.inv(X.T @ X + lambda_para * np.eye(n)) @ X.T @ y
    return beta_ridge

def cross_validation(X, y, lambda_paras, k=10):
    kf = KFold(n_splits=k, shuffle=True, random_state=42)
    mse_score = []

    for lambda_para in lambda_paras:
        mse_list = []
        for train_idx, val_idx in kf.split(X):
            X_train, X_val = X[train_idx], X[val_idx]
            y_train, y_val = y[train_idx], y[val_idx]
            beta = ridge(X_train, y_train, lambda_para)
            
            y_pred = X_val @ beta
            mse = np.mean((y_pred - y_val) ** 2)
            mse_list.append(mse)
        
        avg_mse = np.mean(mse_list)
        mse_score.append(avg_mse)
        print(f" = {lambda_para:.4f}, MSE = {avg_mse:.4f}")
    
    plt.figure(figsize=(10, 6))
    plt.plot(lambda_paras, mse_score, 'o-')
    plt.xlabel(' (log scale)')
    plt.ylabel('avg_MSE')
    plt.title('The ralationship between  and MSE')
    plt.savefig('lambda_vs_mse.png')
    plt.show()
    
    best_idx = np.argmin(mse_score)
    best_lambda = lambda_paras[best_idx]
    print(f"Best : {best_lambda:.4f}")
    
    return best_lambda
